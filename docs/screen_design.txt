        // Micro-interactions for adding a bank
        const addBtn = document.getElementById('addBankBtn');
        const input = document.getElementById('bankInput');
        const list = document.getElementById('bankList');

        addBtn.addEventListener('click', () => {
            const val = input.value.trim();
            if (val) {
                const newItem = document.createElement('div');
                newItem.className = "flex items-center justify-between p-4 bg-white border border-outline-variant rounded-xl shadow-[0_1px_3px_0_rgba(0,0,0,0.05)] hover:shadow-md transition-all opacity-0 scale-95";
                newItem.innerHTML = `
                    <div class="flex items-center gap-4">
                        <div class="w-10 h-10 rounded-full bg-surface-container-high flex items-center justify-center text-primary">
                            <span class="material-symbols-outlined">account_balance</span>
                        </div>
                        <div>
                            <p class="font-headline-sm text-headline-sm leading-tight">${val}</p>
                            <p class="font-label-bold text-label-bold text-on-surface-variant">新增銀行</p>
                        </div>
                    </div>
                    <div class="flex items-center gap-2">
                        <button class="p-2 text-on-surface-variant hover:text-primary hover:bg-surface-container transition-colors rounded-lg">
                            <span class="material-symbols-outlined">edit</span>
                        </button>
                        <button class="p-2 text-on-surface-variant hover:text-error hover:bg-error-container transition-colors rounded-lg" onclick="this.closest('.flex.items-center.justify-between').remove()">
                            <span class="material-symbols-outlined">delete</span>
                        </button>
                    </div>
                `;
                list.prepend(newItem);

                // Animation frame for entrance
                requestAnimationFrame(() => {
                    newItem.classList.remove('opacity-0', 'scale-95');
                    newItem.classList.add('opacity-100', 'scale-100');
                });

                input.value = '';
                input.focus();
            }
        });

        // Trigger on Enter
        input.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') addBtn.click();
        });
    </script>

        // Simple logic for auto-calculating dates and interest
        const amountInput = document.getElementById('amount');
        const rateInput = document.getElementById('rate');
        const periodVal = document.getElementById('period_val');
        const periodUnit = document.getElementById('period_unit');
        const startDateInput = document.getElementById('start_date');
        const endDateInput = document.getElementById('end_date');
        const interestInput = document.getElementById('interest_calc');

        // Set default start date to today
        const today = new Date().toISOString().split('T')[0];
        startDateInput.value = today;

        function updateCalculations() {
            const amount = parseFloat(amountInput.value) || 0;
            const rate = parseFloat(rateInput.value) || 0;
            const period = parseInt(periodVal.value) || 0;
            const unit = periodUnit.value;
            const start = new Date(startDateInput.value);

            if (isNaN(start.getTime())) return;

            // Date Calculation
            let end = new Date(start);
            if (unit === 'days') end.setDate(end.getDate() + period);
            if (unit === 'weeks') end.setDate(end.getDate() + (period * 7));
            if (unit === 'months') end.setMonth(end.getMonth() + period);
            if (unit === 'years') end.setFullYear(end.getFullYear() + period);

            endDateInput.value = end.toISOString().split('T')[0];

            // Simplified Interest Calculation (Actual/365)
            const diffTime = Math.abs(end - start);
            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
            const interest = amount * (rate / 100) * (diffDays / 365);

            interestInput.value = interest.toFixed(2);
        }

        [amountInput, rateInput, periodVal, periodUnit, startDateInput].forEach(el => {
            el.addEventListener('input', updateCalculations);
        });

        document.getElementById('saveBtn').addEventListener('click', () => {
            const btn = document.getElementById('saveBtn');
            btn.innerHTML = '<span class="material-symbols-outlined animate-spin">sync</span>';
            setTimeout(() => {
                btn.innerHTML = '儲存成功！';
                btn.classList.replace('bg-primary', 'bg-secondary');
            }, 1000);
        });

        // Initialize
        updateCalculations();
    </script>
